/**
 * Winky Robot Firmware
 * Hardware: Arduino Uno + L298N Motor Driver + IR Sensor
 */

// Define Pins for L298N Motor Driver
const int IN1 = 9;   // Left Motor direction 1
const int IN2 = 10;  // Left Motor direction 2
const int IN3 = 11;  // Right Motor direction 1
const int IN4 = 12;  // Right Motor direction 2

// Define Pins for Accessories
const int LED_PIN = 13;   // Headlights
const int IR_SENSOR = 2;  // IR Obstacle Sensor

void setup() {
  // Start Serial communication at 9600 baud
  Serial.begin(9600);

  // Set all motor pins as outputs
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);
  
  // Set accessory pins
  pinMode(LED_PIN, OUTPUT);
  pinMode(IR_SENSOR, INPUT);

  // Ensure robot starts in "Stopped" state
  stopRobot();
}

void loop() {
  // --- SAFETY LAYER ---
  // If IR sensor detects something (LOW), override everything and stop
  if (digitalRead(IR_SENSOR) == LOW) {
    stopRobot();
  }

  // --- COMMUNICATION LAYER ---
  // Check if the Web Agent sent a command via USB
  if (Serial.available() > 0) {
    char command = Serial.read(); // Read the character sent (e.g., 'F')

    switch (command) {
      case 'F': // Forward
        moveForward();
        break;
      case 'L': // Forward Left (Slow)
        turnLeftSlow();
        break;
      case 'S': // Stop
        stopRobot();
        break;
      case 'H': // Toggle Lights
        digitalWrite(LED_PIN, !digitalRead(LED_PIN));
        break;
    }
  }
}

// --- MOVEMENT FUNCTIONS ---

void moveForward() {
  // Left Motor Forward (IN1 LOW, IN2 HIGH)
  digitalWrite(IN1, LOW);  digitalWrite(IN2, HIGH);
  // Right Motor Forward (IN3 HIGH, IN4 LOW)
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void turnLeftSlow() {
  // Stop Left Wheel, keep Right Wheel moving forward
  digitalWrite(IN1, LOW);  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void stopRobot() {
  // All motor pins LOW = no movement
  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}
